### Task 2: Add scroll-triggered animations to homepage sections

**Files Modified:** `src/components/sections/ScrollReveal.tsx` (new)

**Action:**
Create a reusable `ScrollReveal` client component using Framer Motion's `useInView`:

```tsx
"use client";

import { useRef } from "react";
import { motion, useInView } from "framer-motion";

interface ScrollRevealProps {
  children: React.ReactNode;
  className?: string;
  delay?: number;
}

export function ScrollReveal({ children, className, delay = 0 }: ScrollRevealProps) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once: true, margin: "-80px" });

  return (
    <motion.div
      ref={ref}
      className={className}
      initial={{ opacity: 0, y: 32 }}
      animate={isInView ? { opacity: 1, y: 0 } : { opacity: 0, y: 32 }}
      transition={{ duration: 0.65, delay, ease: [0.22, 1, 0.36, 1] }}
    >
      {children}
    </motion.div>
  );
}
```

**Acceptance Criteria:**
- `src/components/sections/ScrollReveal.tsx` exists
- File contains `"use client"` directive
- File contains `useInView` from `framer-motion`
- File exports `ScrollReveal` function
