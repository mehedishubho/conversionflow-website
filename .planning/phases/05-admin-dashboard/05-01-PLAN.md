async function requireAdmin() {
  const session = await auth.api.getSession({ headers: await headers() });
  if (!session) redirect("/login");
  const role = (session.user as Record<string, unknown>).role as string;
  if (role !== "admin" && role !== "super_admin") redirect("/admin/dashboard");
  return { session, userId: session.user.id, role };
}

    - Call `requireAdmin()`.
    - Query 4 KPIs for current period + 4 for previous period (8 total queries, batched):
      1. **Total Revenue**: `COALESCE(SUM(amount), 0)` from orders where `status = "completed"` AND `createdAt` between start and now.
      2. **MRR** (per D-04): `COALESCE(SUM(amount), 0)` from orders where `status = "completed"` AND `createdAt` >= start of current calendar month (ignore range param for MRR -- always current month).
      3. **Active Customers**: `COUNT(DISTINCT userId)` from orders where `status = "completed"`.
      4. **Total Orders**: `COUNT(*)` from orders where `createdAt` between start and now.
    - Calculate trend percentage for each: `(current - previous) / previous * 100`. Guard: if previous === 0, show +100% if current > 0, else "N/A" (per RESEARCH Pitfall 4).
    - Return object: `{ totalRevenue, revenueTrend, mrr, mrrTrend, activeCustomers, customersTrend, totalOrders, ordersTrend }` where each is `{ value: number, trend: number, trendDirection: "up" | "down" | "flat" }`.
